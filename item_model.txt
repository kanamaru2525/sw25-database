
// --- 統合Itemモデル ---
model Item {
  id            String          @id @default(cuid())
  itemType      ItemType
  name          String
  category      String?
  rank          Rank?
  usage         String?
  minStrength   Int?
  
  // 武器専用
  hit           Int?
  power         Int?
  critical      Int?
  extraDamage   Int?
  range         Int?
  
  // 防具専用
  evasion       Int?
  defense       Int?
  
  // 共通
  price         Int
  summary       String          @db.Text
  page          String
  regulation    RegulationType
  
  @@index([itemType])
  @@index([regulation])
}
